### 2、后续的确定挂号业务

1. 定义订单接口ts的类型

```ts
// src/apis/user/type.ts

export interface ResponseData {
  code: number
  message: string
  ok: boolean
}

// 最外层嵌套了一层data
export interface OuterResponse<T extends ResponseData> {
  data: T
}

// 提交订单接口返回的数据ts类型
export interface SubmitOrder extends ResponseData {
  data: number
}
```

2. 定义订单号码接口

```ts
// src/apis/user/index.ts

// 引入二次封装的axios
import request from '@/utils/request'
import type { SubmitOrder, OuterResponse } from './type'

enum API {
  // 获取订单号码接口
  SUBMITORDER_URL = '/order/orderInfo/auth/submitOrder/'
}

export const reqSubmitOrder = (
  hoscode: string,
  scheduleId: string,
  patientId: number
) =>
  request.post<any, OuterResponse<SubmitOrder>>(
    API.SUBMITORDER_URL + `${hoscode}/${scheduleId}/${patientId}`
  )
```

3. 给确认按钮回调

```vue
// src/views/hospital/register/register_step2.vue

<script setup lang="ts">
import type { SubmitOrder, OuterResponse } from '@/apis/user/type'
  
// 确定挂号按钮回调
const submitOrder = async () => {
  let hoscode = docInfo.value.hoscode
  let scheduleId = docInfo.value.id
  let patientId = userArr.value[currentIndex.value].id
  // 提交订单
  let result: OuterResponse<SubmitOrder> = await reqSubmitOrder(
    hoscode,
    scheduleId,
    patientId
  )
  if (result.data.code == 200) {
    $router.push({ path: '/user/order', query: { orderId: result.data.data } }
  } else {
    ElMessage({
      type: 'error',
      message: result.data.message
    })
  }
}
</script>

<template>
	<div class="btn">
    <el-button @click="submitOrder">确认挂号</el-button>
  </div>
</template>
```

<script setup lang="ts">
import { watch } from 'vue'
  
const changeScene = (user: AddOrUpdateUser) => {
  ...
  Object.assign(userParams, user)
}

watch(
  () => userArr.value,
  () => {
    if ($route.query.type == 'edit') {
